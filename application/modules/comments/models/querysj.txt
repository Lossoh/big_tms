select DISTINCT f.destination_name as DARI,  d.destination_name as KE, e.vessel_name, a.sj_date from fx_trx_sj a left Join fx_trx_document_separate b ON a.document_id=b.document_id and a.document_separate_id=b.document_separate_id LEFT JOIN fx_trx_vessel_document c ON b.document_id=c.document_id LEFT JOIN fx_mst_destinations d ON a.destination_to= d.destination_id left Join fx_mst_vessels e  ON e.vessel_id=a.vessel_id  LEFT JOIN fx_mst_destinations f on a.destination_from=f.destination_id where a.date_created between '2015-02-25' and '2015-03-31' and a.deleted=0 order by vessel_name,sj_date asc
